      const wrap = document.getElementById('ba');
      const before = document.getElementById('before');
      const handle = document.getElementById('handle');
      const knob = document.getElementById('knob');
      let dragging = false;

      function setX(clientX){
        const rect = wrap.getBoundingClientRect();
        let x = Math.max(0, Math.min(clientX - rect.left, rect.width));
        const p = x / rect.width * 100;
        before.style.clipPath = `inset(0 ${100-p}% 0 0)`;
        handle.style.left = x + 'px';
        knob.style.left = x + 'px';
        knob.setAttribute('aria-valuenow', Math.round(p));
      }

      const onMove = e => setX((e.touches? e.touches[0].clientX : e.clientX));

      ['mousedown','touchstart'].forEach(evt=> knob.addEventListener(evt, ()=> dragging=true));
      ['mouseup','touchend','mouseleave'].forEach(evt=> wrap.addEventListener(evt, ()=> dragging=false));
      wrap.addEventListener('mousemove', e=> dragging && onMove(e));
      wrap.addEventListener('touchmove', onMove, {passive:true});
      wrap.addEventListener('click', onMove);
      // init
      setX(wrap.getBoundingClientRect().width/2 + wrap.getBoundingClientRect().left);

    // Estimator logic (very simple demo model)
    const gbp = new Intl.NumberFormat('en-GB', {style:'currency', currency:'GBP'});
    const size = document.getElementById('size');
    const finish = document.getElementById('finish');
    const ward = document.getElementById('ward');
    const appliances = document.getElementById('appliances');
    const priceEl = document.getElementById('price');
    const sizeOut = document.getElementById('sizeOut');
    const wardOut = document.getElementById('wardOut');

    function calc(){
      const kSize = +size.value; // m¬≤
      const finishMult = +finish.value;
      const wardrobeM = +ward.value;
      const appliancePack = +appliances.value;
      sizeOut.value = kSize;
      wardOut.value = wardrobeM;
      // Base: ¬£650 per m¬≤ for cabinetry + worktops (demo), adjusted by finish
      let kitchen = kSize * 650 * finishMult;
      // Wardrobes: ¬£450 per linear metre
      let wardrobes = wardrobeM * 450;
      // Fitting & sundries 12%
      let fit = (kitchen + wardrobes) * 0.12;
      // Appliances package
      let total = kitchen + wardrobes + fit + appliancePack;
      // VAT 20%
      total *= 1.2;
      priceEl.textContent = gbp.format(Math.round(total));
    }
    ['input','change'].forEach(ev=>{
      size.addEventListener(ev, calc);
      finish.addEventListener(ev, calc);
      ward.addEventListener(ev, calc);
      appliances.addEventListener(ev, calc);
    });
    calc();

    // Testimonials carousel
    (function(){
      const slides = [...document.querySelectorAll('.slide')];
      const dots = [...document.querySelectorAll('.dots button')];
      let i = 0; let timer;
      function show(n){
        i = (n+slides.length)%slides.length;
        slides.forEach((s,idx)=> s.classList.toggle('active', idx===i));
        dots.forEach((d,idx)=> d.setAttribute('aria-current', idx===i ? 'true' : 'false'));
      }
      function auto(){ timer = setInterval(()=>{ show(i+1); }, 4500); }
      dots.forEach((d,idx)=> d.addEventListener('click', ()=>{ clearInterval(timer); show(idx); auto(); }));
      show(0); auto();
    })();
